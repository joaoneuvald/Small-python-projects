while not end:
  guess = input("Choose a letter: ").lower()

  if guess in blanksheet:
    print(f"You've already guessed {guess}.")
  
  for position in range(len(hang_word)):
    letter = hang_word[position]
    if  letter == guess:
      blanksheet[position] = letter
      print(blanksheet)
      
  if guess not in hang_word:
    lives -=1
    print(stages[lives])
    print(f"There is no {guess} in this word.")
    print(blanksheet)
    if lives == 0:
      print("You lose")
      end = True

  if '_' not in blanksheet:
    end = True
    print("You win")
